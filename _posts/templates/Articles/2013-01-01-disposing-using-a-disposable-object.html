Disposing using a disposable object

The method is used to register a disposable item. A disposable item is simply an object with a dispose function

var myViewModel = wo.view.extend({ constructor: function myViewModel() { this._super(); }, onModelChanged: function(oldModel, newModel) { this._super(oldModel, newModel); if(newModel) { var eventRegistration = newModel.myObservable.subscribe(function() { // do something }); // eventRegistration will be of the form { dispose: function() { } } this.registerDisposeCallback(eventRegistration); } } });
